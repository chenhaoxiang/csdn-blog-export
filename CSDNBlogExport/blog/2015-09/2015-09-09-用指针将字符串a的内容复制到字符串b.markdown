/**下面用指针来处理这个问题**/
int main()
{
    char a[]="i love you!",b[100];
    char *p1,*p2;
    p1=a;p2=b;
    for(;*p1!='\0';p1++,p2++)
        *p2=*p1;
    *p2='\0';
    printf("a:%s\n",a);
    printf("b:%s\n",b);
    return 0;
}
