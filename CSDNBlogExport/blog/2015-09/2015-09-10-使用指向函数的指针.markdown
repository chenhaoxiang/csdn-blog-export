/**输入2个整数，然后让用户选择1或2，选1时调用max函数，输出2者中的大数，
选2时调用min函数，输出2者中的小数**/
#include <stdio.h>
#include <stdlib.h>

int main()
{
    int max(int x,int y);
    int min(int x,int y);
    int (*p)(int ,int );
    int n,a,b;
    scanf("%d%d",&a,&b);
    scanf("%d",&n);
    if(n==1)
    {
        p=max;
    }
    else p=min;
    int c=(*p)(a,b);
    printf("%d\n",c);
    return 0;
}
int max(int x,int y)
{
    int z;
    if(x>y)
        z=x;
    else z=y;
    return z;
}
int min(int x,int y)
{
    int z;
    if(x<y)
        z=x;
    else
        z=y;
    return z;
}
