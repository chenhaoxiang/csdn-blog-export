#include <stdio.h>
#include <stdlib.h>

void check(int *p[])
{
    int i;
    printf("没及格的成绩有：\n");
    for(i=0;i<5;i++)
    {
        if(p[i]<60)
            printf("%d ",p[i]);
    }
        printf("\n");
}
int main()
{
    int *pl,i;
    pl=(int *)malloc(5*sizeof(int));
    for(i=0;i<5;i++)
    {
        scanf("%d",pl+i);
    }
    check(pl);
    return 0;
}
